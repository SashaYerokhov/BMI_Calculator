Чтобы вывести результат из двух input без кнопки Submit, нужно отслеживать событие ввода (например, input) в обоих полях и сразу же обновлять область вывода с помощью JavaScript.

 <input type="number" id="num1" placeholder="Число 1">
<input type="number" id="num2" placeholder="Число 2">

<p>Результат: <span id="result">0</span></p>

<script>
  // Находим элементы на странице
  const input1 = document.getElementById('num1');
  const input2 = document.getElementById('num2');
  const resultOutput = document.getElementById('result');

  // Функция для обновления результата
  function updateResult() {
    const val1 = parseFloat(input1.value) || 0;
    const val2 = parseFloat(input2.value) || 0;
    resultOutput.textContent = val1 + val2;
  }

  // Добавляем слушатели событий для обоих полей
  input1.addEventListener('input', updateResult);
  input2.addEventListener('input', updateResult);
</script>

Как это работает:Событие input срабатывает моментально при каждом нажатии клавиши или вставке текста.Функция updateResult считывает текущие значения. Метод parseFloat() превращает их в числа, а || 0 защищает от ошибок, если поле пустое.Результат сразу записывается в тег <span>.Если вам нужно не сложение, а просто объединение (конкатенация) строк, замените строку в функции на:resultOutput.textContent = input1.value + input2.value;

  function calculationBmi() {
  const root = document.querySelector(".form");
  if (!root) return;

  const selectSystems = root.querySelectorAll('input[name="radio"]');
  const optionsSystems = root.querySelectorAll(".header__input");

  selectSystems.forEach((select) => {
    select.addEventListener("change", (event) => {
      const show = select.dataset.show;

      optionsSystems.forEach((options) => {
        options.hidden = true;
        const id = options.getAttribute("id");
        if (id === show) {
          options.hidden = false;
        }
      });
      checkBmiInputs();
    });
  });

  const inputHeight = root.querySelector("#inputHeight");
  const inputWeight = root.querySelector("#inputWeight");
  const result = root.querySelector("#result");
  const recommendationWeight = root.querySelector("#recommendation-weight");
  // const bmiResults = root.querySelectorAll(".bmi-result");
  // Находим блоки результатов
  const resultEmpty = root.querySelector(".bmi-result.empty");
  const resultFull = root.querySelector(".bmi-result.full");

  // Функция проверки заполнения полей
  function checkBmiInputs() {
    // Находим контейнер, который сейчас НЕ скрыт (активен)
    const activeContainer = root.querySelector(".header__input:not([hidden])");

    if (!activeContainer) return;

    // Берем инпуты только из активного контейнера
    const activeInputs = activeContainer.querySelectorAll("input");

    // Проверяем, что КАЖДОЕ активное поле заполнено (длина текста > 0)
    const allFilled = Array.from(activeInputs).every(
      (input) => input.value.trim() !== "",
    );

    if (allFilled) {
      // Показываем блок с результатом
      resultEmpty.style.display = "none";
      resultFull.classList.add("active"); // Используем ваш класс active
      resultFull.style.display = "block";
    } else {
      // Возвращаем приветственный блок
      resultEmpty.style.display = "block";
      resultFull.classList.remove("active");
      resultFull.style.display = "none";
    }
  }

  // Навешиваем обработчик на ВСЕ инпуты в обоих блоках
  const allInputs = document.querySelectorAll(".header__input input");
  allInputs.forEach((input) => {
    input.addEventListener("input", checkBmiInputs);
  });

  // Дополнительно: вызываем проверку при смене вкладок метрическая/имперская
  // (Добавьте вызов функции checkBmiInputs() в ваш код переключения hidden между metric и imperial)

  function calculationBmi() {
    const heightValue = parseFloat(inputHeight.value);
    const weightValue = parseFloat(inputWeight.value);

    if (!heightValue || isNaN(heightValue) || heightValue < 0) {
      return;
    }
    if (!weightValue || isNaN(weightValue) || weightValue < 0) {
      return;
    }

    const bmi = (weightValue / ((heightValue * heightValue) / 10000)).toFixed(
      2,
    );
    result.innerText = `${bmi}`;
    if (heightValue <= 150) {
      recommendationWeight.innerText = `45kgs-58kgs`;
    } else if (heightValue >= 155 && heightValue < 160) {
      recommendationWeight.innerText = `50kgs-63kgs`;
    } else if (heightValue >= 160 && heightValue < 165) {
      recommendationWeight.innerText = `52kgs-66kgs`;
    } else if (heightValue >= 165 && heightValue < 170) {
      recommendationWeight.innerText = `54kgs-69kgs`;
    } else if (heightValue >= 170 && heightValue < 175) {
      recommendationWeight.innerText = `56kgs-72kgs`;
    } else if (heightValue >= 175 && heightValue < 180) {
      recommendationWeight.innerText = `59kgs-75kgs`;
    } else if (heightValue >= 180 && heightValue < 185) {
      recommendationWeight.innerText = `61kgs-78kgs`;
    } else if (heightValue >= 185 && heightValue < 190) {
      recommendationWeight.innerText = `64kgs-82kgs`;
    } else if (heightValue >= 190) {
      recommendationWeight.innerText = `66kgs-84kgs`;
    }
  }

  inputHeight.addEventListener("input", calculationBmi);
  inputWeight.addEventListener("input", calculationBmi);
}
